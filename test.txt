void	check_input(t_data *data)
{
	if (data->here_doc)
	{
		if (data->argc < 6)
		{
			perror("Usage: ./pipex here_doc LIMITER cm1 cmd2 ... file");
			exit(EXIT_FAILURE);
		}
		data->cmd_nbrs = data->argc - 4;
	}
	else
	{
		if (data->argc < 5)
		{
			perror("Usage: ./pipex infile cmd1 cmd2 ... outfile");
			exit(EXIT_FAILURE);
		}
		data->cmd_nbrs = data->argc - 3;
	}
}
